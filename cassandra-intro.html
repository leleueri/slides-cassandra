CQL / Skinny row


create table skinny (
    userid varchar primary key, 
    firstname varchar, 
    lastname varchar, 
    birth_year int);

Primary Key == Partition Key (== Row Key)

CQL / Skinny row


insert into skinny (userid, firstname, lastname, birth_year) 
values ('user0001', 'f_user0001', 'l_user0001', 1983);

CQL / Skinny row


$>select * from skinny; 

 userid   | birth_year | firstname     | lastname
----------+------------+---------------+-------------

 user0002 |       null | f_user0002    | l_user0002

(2 rows)

Where clause on primary key


$>select * from skinny where userid = 'user0001'; 

 userid   | birth_year | firstname     | lastname

CQL / Wide row


create table wide (objectid varchar, resolution varchar, 
    format varchar, size int, 
    primary key (objectid, resolution));
  • Primary Key == (Partition Key + Clustering Key)
  • Partition Key == Row Key

    • CQL / Wide row

    
$> select * from wide;

 objectid  | resolution | format | size
-----------+------------+--------+--------
 object002 |    800x600 |    PNG | 10240
 object001 |    128x128 |   JPEG |  1024
 object001 |    800x600 |   JPEG | 10240
 
(3 rows)

    CQL / Wide row

    
$> select * from wide where objectid = 'object001';

 objectid  | resolution | format | size
-----------+------------+--------+--------
 object001 |    128x128 |   JPEG |  1024
 object001 |    800x600 |   JPEG | 10240
 
(2 rows)
					
    
$> select * from wide where objectid = 'object001' 
    and resolution > '128x128';

 objectid  | resolution | format | size
-----------+------------+--------+--------
 object001 |    800x600 |   JPEG | 10240
