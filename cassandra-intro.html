CQL / Skinny partition


create table skinny (
    userid varchar primary key, 
    firstname varchar, 
    lastname varchar, 
    email varchar);

  • Primary Key == Partition Key
  • No mandatory columns

CQL / Skinny partition


insert into skinny (userid, firstname, lastname, email) 
values ('user0001', 'John', 'Doe', "jdoe@yahoo.fr");
                    

update skinny set email = 'jdoe@gmail.com' where userid = 'user0001';

CQL / Wide partition


create table wide (objectid varchar, hour long, 
    temp float, pressure float,
    primary key (objectid, hour));
  • Primary Key == (Partition Key + Clustering Key)

    • CQL / Wide partition

    
$> select * from wide;

 objectid  | hour            | temp    | pressure
-----------+-----------------+---------+----------
 object002 |    201811091600 |    24.2 | 1024.0
 object001 |    201811091600 |    22.6 | 1022.5
 object001 |    201811091700 |    22.2 | 1022.7
 
(3 rows)

    CQL / Wide partition

    
$> select * from wide where objectid = 'object001';

 objectid  | hour            | temp    | pressure
-----------+-----------------+---------+----------
 object001 |    201811091600 |    22.6 | 1022.5
 object001 |    201811091700 |    22.2 | 1022.7
 
(2 rows)

    CQL / Wide partition

    
$> select * from wide where objectid = 'object001' 
    and hour > 201811091600;

 -----------+-----------------+---------+----------
